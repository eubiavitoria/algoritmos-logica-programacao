Algoritmo "Viagem_Copa_2026"

Var
   valor: real
   falta: real

Inicio
   escreva("Digite o valor acumulado em conta: ")
   leia(valor)

   se valor >= 25000 entao
      escreva("Partiu America do Norte")
   senao
      falta <- 25000 - valor
      escreva("Faltam R$ ", falta, " para atingir a meta.")
   fimse
Fimalgoritmo
