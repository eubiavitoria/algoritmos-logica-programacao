Algoritmo "Simulador_CDB"

Var
   valor: real
   prazo: inteiro
   taxa: real

Inicio
   escreva("Digite o valor do investimento: ")
   leia(valor)

   escreva("Digite o prazo em meses: ")
   leia(prazo)

   se prazo > 12 entao
      taxa <- 12
   senao
      taxa <- 8
   fimse

   escreva("A taxa aplicada sera de ", taxa, "% ao ano.")
Fimalgoritmo
